-	script	comptage	-1,{ // script when the Player count until 10
OnStart:
	set $@seek1$,rand(0,24);  // on fait les randoms pour définir quel NPC sera où
	goto L_Seek2;
	end;

L_Seek2:
	set $@seek2$,rand(0,24);
	if($@seek2$==$@seek1$) goto L_Seek2; // avec les vérifications habituelles pour éviter d'en avoir 2 au même endroit
	else;
	goto L_Seek3;
	end;

L_Seek3:
	set $@seek3$,rand(0,24);
	if($@seek3$==$@seek2$ || $@seek3$==$@seek1$) goto L_Seek3;
	else;
	goto L_Seek4;
	end;

L_Seek4:
	set $@seek4$,rand(0,24);
	if($@seek4$==$@seek3$ || $@seek4$==$@seek2$ || $@seek4$==$@seek1$) goto L_Seek4;
	else;
	goto L_Seek5;
	end;

L_Seek5:
	set $@seek5$,rand(0,24);
	if($@seek5$==$@seek4$ || $@seek5$==$@seek3$ || $@seek5$==$@seek2$ || $@seek5$==$@seek1$) goto L_Seek5;
	else;
	initnpctimer;
	end;

OnTimer1000:  // then, "Famous Five" go to hide themselves
	disablenpc "nothidden1"; // gone
	enablenpc $@array1$[atoi($@seek1$)]; // hidden
	mapannounce "lighthalzen","[Famous Five] : 1.",1;
	end;

OnTimer2000:
	mapannounce "lighthalzen","[Famous Five] : 2.",1;
	end;

OnTimer3000:
	disablenpc "nothidden2"; // gone
	enablenpc $@array2$[atoi($@seek2$)]; // hidden
	mapannounce "lighthalzen","[Famous Five] : 3.",1;
	end;

OnTimer4000:
	mapannounce "lighthalzen","[Famous Five] : 4.",1;
	end;
	
OnTimer5000:
	disablenpc "nothidden3"; // gone
	enablenpc $@array3$[atoi($@seek3$)]; // hidden
	mapannounce "lighthalzen","[Famous Five] : 5.",1;
	end;

OnTimer6000:
	mapannounce "lighthalzen","[Famous Five] : 6.",1;
	end;

OnTimer7000:
	disablenpc "nothidden4"; // gone
	enablenpc $@array4$[atoi($@seek4$)]; // hidden
	mapannounce "lighthalzen","[Famous Five] : 7.",1;
	end;

OnTimer8000:
	mapannounce "lighthalzen","[Famous Five] : 8.",1;
	end;

OnTimer9000:
	disablenpc "nothidden5"; // gone
	enablenpc $@array5$[atoi($@seek5$)]; // hidden
	mapannounce "lighthalzen","[Famous Five] : 9.",1;
	end;

OnTimer10000: // GO !
	mapannounce "lighthalzen","[Famous Five] : 10 !! It's OK now. You can look for us.",1;
	end;

OnTimer3600000: // after one hour, if the Player didn't find them all, it's over
	mapannounce "lighthalzen","[Famous Five] : Time Out ! You lost ! See you at the meeting point",1;
	disablenpc $@array1$[atoi($@seek1$)];
	disablenpc $@array2$[atoi($@seek2$)];
	disablenpc $@array3$[atoi($@seek3$)];
	disablenpc $@array4$[atoi($@seek4$)];
	disablenpc $@array5$[atoi($@seek5$)];
	enablenpc "nothidden1";
	enablenpc "nothidden2";
	enablenpc "nothidden3";
	enablenpc "nothidden4";
	enablenpc "nothidden5";
	set $@seek,0;
	set $@timeout,1;
	set $@cherche,0;
	stopnpctimer;
	end;
}
